1function I_recover1 = decompression_hdl(encoded_bitstream,height,width,I_truth,subimage_index,I_subimages)
  2
  3pixel_index = 1;     % point to the pixel to be processed
  4bitstream_index = 1; % point to the next bit
  5bitstream_length = length(encoded_bitstream);
  6I_recover = zeros(height*width,1);
  7range = 8;
  8I_truth_vector = I_truth';
  9I_truth_vector = I_truth_vector(:);
 10residue_learned = load('residue_learned1.mat');
 11residue_learned = int(residue_learned.residue_learned1);
 12
 13mode = 0;
 14while (bitstream_index <= bitstream_length)
 15    if (mode == 0)
 16        % context calculation
 17        [a,b,c,d,e] = context_calculation_zeropad(I_recover, width, pixel_index);
 18        xx = ceil(pixel_index/width);
 19        yy = mod(pixel_index,width);
 20        if yy == 0
 21            yy = width;
 22        end
 23        g1 = a - c; g2 = c - b; g3 = d - a; g4 = b - e;
 24        
 25        upperbound = 255;
 26        lowerbound = 0;
 27        switch subimage_index
 28            case 1
 29                gq1 = double(g_quantize(g1));
 30                gq2 = double(g_quantize(g2));
 31                gq3 = double(g_quantize(g3));
 32                gq4 = double(g_quantize(g4));
 33                context_index = gq1*5^3 + gq2*5^2 + gq3*5^1 + gq4 + 1;
 34                residue = residue_learned(context_index);
 35                if residue == int8(8)
 36                    if(c >= max(b,a))
 37                        predict_x = min(b,a);
 38                    elseif(c <= min(b,a))
 39                        predict_x = max(b,a);
 40                    else
 41                        predict_x = b + a - c;
 42                    end
 43                else
 44                    predict_x = b + residue;
 45                    if predict_x < 0
 46                        predict_x = 0;
 47                    end
 48                    if predict_x > 7
 49                        predict_x = 7;
 50                    end
 51                end
 52            case 2 % inter-prediction
 53                neighbor1 = subimage_access(I_subimages{1},xx,yy);
 54                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
 55                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(4);
 56                predict_x0 = uint8(predict_x0);
 57                predict_x = int8(bitsrl(predict_x0, 5));
 58            case 3
 59                neighbor1 = subimage_access(I_subimages{1},xx,yy+1);
 60                neighbor2 = subimage_access(I_subimages{2},xx,yy);
 61                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
 62                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
 63                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(7);
 64                predict_x0 = uint8(predict_x0);
 65                predict_x = int8(bitsrl(predict_x0, 5));
 66            case 4
 67                neighbor1 = subimage_access(I_subimages{1},xx,yy+1);
 68                neighbor2 = subimage_access(I_subimages{2},xx,yy);
 69                neighbor3 = subimage_access(I_subimages{3},xx,yy);
 70                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
 71                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
 72                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor3, int16(7));
 73                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(11);
 74                predict_x0 = uint8(predict_x0);
 75                predict_x = int8(bitsrl(predict_x0, 5));
 76            case 5
 77                neighbor1 = subimage_access(I_subimages{1},xx,yy);
 78                neighbor2 = subimage_access(I_subimages{2},xx,yy);
 79                neighbor3 = subimage_access(I_subimages{3},xx,yy);
 80                neighbor4 = subimage_access(I_subimages{4},xx,yy);
 81                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
 82                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
 83                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor3, int16(7));
 84                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor4, int16(11));
 85                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(14);
 86                predict_x0 = uint8(predict_x0);
 87                predict_x = int8(bitsrl(predict_x0, 5));
 88            case 6
 89                neighbor1 = subimage_access(I_subimages{1},xx,yy);
 90                neighbor2 = subimage_access(I_subimages{2},xx,yy);
 91                neighbor3 = subimage_access(I_subimages{3},xx,yy-1);
 92                neighbor4 = subimage_access(I_subimages{4},xx,yy-1);
 93                neighbor5 = subimage_access(I_subimages{5},xx,yy);
 94                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
 95                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
 96                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor3, int16(7));
 97                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor4, int16(11));
 98                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor5, int16(14));
 99                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(18);
100                predict_x0 = uint8(predict_x0);
101                predict_x = int8(bitsrl(predict_x0, 5));
102            case 7
103                neighbor1 = subimage_access(I_subimages{1},xx,yy);
104                neighbor2 = subimage_access(I_subimages{2},xx,yy);
105                neighbor3 = subimage_access(I_subimages{3},xx,yy-1);
106                neighbor4 = subimage_access(I_subimages{4},xx,yy-1);
107                neighbor5 = subimage_access(I_subimages{5},xx,yy);
108                neighbor6 = subimage_access(I_subimages{6},xx,yy);
109                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
110                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
111                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor3, int16(7));
112                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor4, int16(11));
113                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor5, int16(14));
114                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor6, int16(18));
115                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(21);
116                predict_x0 = uint8(predict_x0);
117                predict_x = int8(bitsrl(predict_x0, 5));
118            case 8
119                neighbor1 = subimage_access(I_subimages{1},xx,yy);
120                neighbor2 = subimage_access(I_subimages{2},xx,yy);
121                neighbor3 = subimage_access(I_subimages{3},xx,yy);
122                neighbor4 = subimage_access(I_subimages{4},xx,yy);
123                neighbor5 = subimage_access(I_subimages{5},xx,yy);
124                neighbor6 = subimage_access(I_subimages{6},xx,yy);
125                neighbor7 = subimage_access(I_subimages{7},xx,yy);
126                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
127                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
128                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor3, int16(7));
129                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor4, int16(11));
130                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor5, int16(14));
131                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor6, int16(18));
132                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor7, int16(21));
133                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(25);
134                predict_x0 = uint8(predict_x0);
135                predict_x = int8(bitsrl(predict_x0, 5));
136            case 9
137                neighbor1 = subimage_access(I_subimages{1},xx,yy+1);
138                neighbor2 = subimage_access(I_subimages{2},xx,yy);
139                neighbor3 = subimage_access(I_subimages{3},xx,yy);
140                neighbor4 = subimage_access(I_subimages{4},xx,yy);
141                neighbor5 = subimage_access(I_subimages{5},xx,yy);
142                neighbor6 = subimage_access(I_subimages{6},xx,yy+1);
143                neighbor7 = subimage_access(I_subimages{7},xx,yy+1);
144                neighbor8 = subimage_access(I_subimages{8},xx,yy);
145                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor1, int16(0));
146                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor2, int16(4));
147                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor3, int16(7));
148                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor4, int16(11));
149                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor5, int16(14));
150                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor6, int16(18));
151                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor7, int16(21));
152                [upperbound, lowerbound] = bound_update_scalar(upperbound, lowerbound, neighbor8, int16(25));
153                predict_x0 = bitsra((upperbound + lowerbound),1) + int16(28);
154                predict_x0 = uint8(predict_x0);
155                predict_x = int8(bitsrl(predict_x0, 5));
156        end
157                
158        if (xx==1 && yy>1)
159            predict_x = e;
160        end
161        
162        % compute gradients 
163        if(g1 == g2 && g1 == g3 && g1 == g4 && g1 == 0 && pixel_index > 1)  % if uniform patch && not first pixel
164            mode = 1;                        % go to runlength mode
165            run_value = b;
166        end
167    end
168    
169    % predictive mode
170    if (mode == 0)
171        % step out and decode Golomb code
172        error_value = 0;
173        while(encoded_bitstream(bitstream_index) == '1')
174            error_value = error_value + 1;
175            bitstream_index = bitstream_index + 1;
176            if bitstream_index > bitstream_length
177                break;
178            end
179        end
180        bitstream_index = bitstream_index + 1;
181        
182        % map back error
183        error_value_mapped = remap(error_value, predict_x, range);
184        x = error_value_mapped + predict_x;
185        
186        I_recover(pixel_index) = x;
187        pixel_index = pixel_index + 1;
188    end
189    
190    % runlength mode
191    if (mode == 1)        
192        % distance to the end of line
193        width1 = mod(pixel_index,width);
194        if (width1 == 0)
195            width1 = width;
196        end
197        max_length = width - width1 + 1;
198        
199        % step out runlength code
200        run_count = 0;
201        while(encoded_bitstream(bitstream_index) == '1')
202            run_count = run_count + 1;
203            bitstream_index = bitstream_index + 1;
204            if bitstream_index > bitstream_length
205                break;
206            end
207        end
208        bitstream_index = bitstream_index + 1;
209        
210        % assign run values
211        for i = 1:run_count
212            I_recover(pixel_index) = run_value;
213            pixel_index = pixel_index + 1;
214        end
215        
216        % check EOL
217        if (run_count >= max_length)
218            EOL = 1;
219        else
220            EOL = 0;
221        end
222        
223        % step out runvalue code (whether EOL)
224        if (~EOL)
225            [a,b] = context_calculation_zeropad(I_recover, width, pixel_index);
226            if (b==a)
227                run_interrupt_type = 1;
228            else
229                run_interrupt_type = 0;
230            end
231            predict_x = a;
232        
233            run_error_value_mapped = 0;
234            while(encoded_bitstream(bitstream_index) == '1')
235                run_error_value_mapped = run_error_value_mapped + 1;
236                bitstream_index = bitstream_index + 1;
237                if bitstream_index > bitstream_length
238                    break;
239                end
240            end
241            bitstream_index = bitstream_index + 1;
242            
243            % map back runvalue error
244            run_error_value = remap(run_error_value_mapped + run_interrupt_type, predict_x, range);
245            %if((run_interrupt_type == 0) && (b>a))
246            %run_error_value = -run_error_value;
247            %end
248            interrupt_value = run_error_value + predict_x;
249            
250            % assign interrupt value
251            I_recover(pixel_index) = interrupt_value;
252            pixel_index = pixel_index + 1;
253        end
254        
255        % return to predictive mode
256        mode = 0; 
257    end
258    
259end %end of while loop
260
261I_recover1 = reshape(I_recover, [width, height])' * 32;
262% for debugging
263% disp(['decompressing the subimage: ', num2str(subimage_index)])
264% figure;imshow(abs(double(I_recover1)-double(I_truth)),[0,32]);
265end

267function value = subimage_access(subimage,x,y)
268[height,width] = size(subimage);
269if x==0 || y == 0 || x>height || y>width
270    value = 0;
271else
272    value = subimage(x,y);
273end
274end
